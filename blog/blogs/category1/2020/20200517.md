```js
// 过滤emoji表情
function filteremoji (val) {
  var ranges = [
    '\ud83c[\udf00-\udfff]',
    '\ud83d[\udc00-\ude4f]',
    '\ud83d[\ude80-\udeff]'
  ]
  var emojireg = val
  emojireg = emojireg.replace(new RegExp(ranges.join('|'), 'g'), '')
  return emojireg
}
```
