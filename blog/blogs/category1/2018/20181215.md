```JS
// 阻止冒泡
function stopDefault (e) {
  if (e && e.preventDefault) {
    e.preventDefault()
  } else {
    window.event.returnValue = false
  }
  return false
}
stopDefault(e)

function stopBubble (e) {
  if (e && e.stopPropagation) {
    e.stopPropagation()
  } else {
    window.event.cancelBubble = true
  }
}
stopBubble(e)
```
